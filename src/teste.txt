N: 8
SIDE: 40
MARGIN: 5
BLACK: |0|
WHITE: |255|
GRAY: |128|
---
dimension: N*SIDE + MARGIN*2 ~ N*SIDE + MARGIN*2
---
for l in [0,N[ {
  for c in [0,N[ {
    if (l + c) % 2 = 0 {
      fill WHITE
    } else {
      fill BLACK
    }
    square (c * SIDE + MARGIN,l * SIDE + MARGIN) SIDE
  }
}
border 0|0|255
